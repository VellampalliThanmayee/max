# Python program to find the largest 
# number among the three numbers 

def maximum(a, b, c): 

	if (a >= b) and (a >= b): 
		largest = a 

	elif (b >= a) and (b >= a): 
		largest = b 
	else: 
		largest = c 
		
	return largest
